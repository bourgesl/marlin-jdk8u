  47             log.info(msg);

  49             System.out.print("INFO: ");

  51         }
  52     }
  53

  55         if (MarlinConst.useLogger) {
  56 //            log.warning(msg, th);
  57             log.log(java.util.logging.Level.WARNING, msg, th);
  58         } else if (MarlinConst.enableLogs) {
  59             System.out.print("WARNING: ");

  62         }
  63     }
